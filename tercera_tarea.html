import numpy as np
import matplotlib.pyplot as plt
from itertools import combinations
from scipy.spatial import ConvexHull

def graficar_restricciones(restricciones, limites=(0, 20, 0, 20)):
    """
    Grafica restricciones lineales y su regi√≥n factible.

    restricciones : lista de tuplas (a, b, c, tipo)
        a*x + b*y <= c ---> tipo = "<="
        a*x + b*y >= c ---> tipo = ">="
    limites : tupla (xmin, xmax, ymin, ymax)
    """
    xmin, xmax, ymin, ymax = limites
    x = np.linspace(xmin, xmax, 400)
    
    plt.figure(figsize=(7, 7))

    # Dibujar las rectas de las restricciones
    for a, b, c, tipo in restricciones:
        if b != 0:
            y = (c - a * x) / b
            plt.plot(x, y, label=f"{a}x + {b}y {tipo} {c}")
        else:
            x_line = np.full_like(x, c / a)
            plt.plot(x_line, x, label=f"{a}x {tipo} {c}")

    # Ejes
    plt.axhline(0, color="black", linewidth=0.8)
    plt.axvline(0, color="black", linewidth=0.8)

    # Encontrar puntos de intersecci√≥n
    puntos = []
    for (a1, b1, c1, _), (a2, b2, c2, _) in combinations(restricciones, 2):
        det = a1 * b2 - a2 * b1
        if det != 0:  # hay intersecci√≥n
            x_int = (c1 * b2 - c2 * b1) / det
            y_int = (a1 * c2 - a2 * c1) / det
            if xmin <= x_int <= xmax and ymin <= y_int <= ymax:
                puntos.append((x_int, y_int))

    # Agregar esquinas del gr√°fico
    puntos += [(xmin, ymin), (xmin, ymax), (xmax, ymin), (xmax, ymax)]

    # Filtrar puntos factibles
    factibles = []
    for px, py in puntos:
        valido = True
        for a, b, c, tipo in restricciones:
            if tipo == "<=" and not (a * px + b * py <= c + 1e-6):
                valido = False
            if tipo == ">=" and not (a * px + b * py >= c - 1e-6):
                valido = False
        if valido:
            factibles.append((px, py))

    # Regi√≥n factible
    if factibles:
        factibles = np.array(factibles)
        hull = ConvexHull(factibles)
        pts = factibles[hull.vertices]
        plt.fill(pts[:, 0], pts[:, 1], "skyblue", alpha=0.4, label="Regi√≥n factible")

    # Configuraci√≥n final
    plt.xlim(xmin, xmax)
    plt.ylim(ymin, ymax)
    plt.xlabel("x")
    plt.ylabel("y")
    plt.legend()
    plt.grid(True, linestyle="--", alpha=0.6)
    plt.title("Restricciones y Regi√≥n Factible")
    plt.show()


# Ejemplo de uso:
restricciones = [
    (1, 1, 15, "<="),
    (2, 1, 18, "<="),
    (1, 2, 20, "<="),
    (1, 0, 0, ">="),
    (0, 1, 0, ">=")
]

graficar_restricciones(restricciones)
