El m√©todo de la secante permite hallar ra√≠ces de funciones sin necesidad de derivarlas, usando dos puntos iniciales.

üß© Definici√≥n

Se basa en aproximar la derivada por una recta secante entre dos puntos:

‚öôÔ∏è Procedimiento

# M√©todo de la Secante
# f(x) = x^2 - 4

import math

def f(x):
    return x**2 - 4

x0 = 1
x1 = 3
tolerancia = 1e-6
iteracion = 0
max_iter = 100

print(f"{'Iter':<6}{'x_n-1':<12}{'x_n':<12}{'x_{n+1}':<12}{'Error':<12}")
print("-"*55)

while True:
    iteracion += 1
    f0 = f(x0)
    f1 = f(x1)
    x2 = x1 - f1 * (x1 - x0) / (f1 - f0)
    error = abs(x2 - x1)

    print(f"{iteracion:<6}{x0:<12.6f}{x1:<12.6f}{x2:<12.6f}{error:<12.6f}")

    if error < tolerancia or iteracion >= max_iter:
        break

    x0, x1 = x1, x2

print("\nRa√≠z aproximada:", round(x2, 6))
print("Iteraciones realizadas:", iteracion)
