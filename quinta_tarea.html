Este c√≥digo usa el m√©todo de Newton-Raphson para encontrar la ra√≠z de una funci√≥n.

funcion = input("Ingrese la funci√≥n f(x): ")       
derivada = input("Ingrese la derivada f'(x): ")    
x0 = float(input("Ingrese el valor inicial x0: "))

tolerancia = 1e-12
iteracion = 0

while True:
    x = x0   
    f = eval(funcion)
    df = eval(derivada)
    x1 = x0 - f/df
    iteracion += 1
    
    if abs(x1 - x0) < tolerancia:
        break
    
    if iteracion > 1000:
        print("No converge despu√©s de muchas iteraciones.")
        break
    
    x0 = x1

print("Ra√≠z aproximada:", x1)
print("Iteraciones realizadas:", iteracion)
