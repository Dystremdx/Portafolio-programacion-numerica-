f(x) = x^3 - 5x + 1  en el intervalo [0, 1]

def f(x):
    return x**3 - 5*x + 1

def regula_falsi(a, b, tol=1e-6, max_iter=20):
    if f(a) * f(b) >= 0:
        print("El intervalo no es v√°lido.")
        return None

    for i in range(max_iter):
        c = b - (f(b) * (a - b)) / (f(a) - f(b))
        print(f"Iter {i+1}: a={a:.6f}, b={b:.6f}, c={c:.6f}, f(c)={f(c):.6e}")
        if abs(f(c)) < tol:
            print(f"\nRa√≠z aproximada: x = {c:.6f}")
            return c
        if f(a) * f(c) < 0:
            b = c
        else:
            a = c

    print("\nNo se alcanz√≥ la convergencia.")
    return None

a, b = 0, 1
regula_falsi(a, b)
