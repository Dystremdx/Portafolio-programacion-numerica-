El m√©todo de bisecci√≥n permite encontrar una ra√≠z al dividir el intervalo en mitades sucesivas hasta alcanzar la tolerancia deseada.

import math

# Definici√≥n de la funci√≥n
def f(x):
    return math.exp(3*x) - 4

a = 0
b = 1
tolerancia = 0.1

print(f"{'Iter':<6}{'a':<10}{'b':<10}{'m':<10}{'f(a)':<12}{'f(b)':<12}{'f(m)':<12}{'Error':<10}")
print("-"*75)

iteracion = 0
while True:
    iteracion += 1
    m = (a + b) / 2
    fa = f(a)
    fb = f(b)
    fm = f(m)
    error = (b - a) / 2

    print(f"{iteracion:<6}{a:<10.5f}{b:<10.5f}{m:<10.5f}{fa:<12.5f}{fb:<12.5f}{fm:<12.5f}{error:<10.5f}")

    if abs(error) < tolerancia:
        break

    if fa * fm < 0:
        b = m
    else:
        a = m

print("\nRa√≠z aproximada: x =", round(m, 4))
print("Error final:", round(error, 5))
